```mermaid
flowchart TD
    subgraph مشتری
        A[شروع: ورود/ثبت‌نام] --> B(جستجو/انتخاب خدمت);
        B --> C(انتخاب تاریخ و زمان);
        C --> D(مشاهده خلاصه و هزینه);
        D --> E{آیا رزرو نهایی شود؟};
        E -- خیر --> C;
        E -- بله --> F(انتقال به صفحه پرداخت);
    end
    subgraph پرداخت
        F --> G(انجام پرداخت آنلاین);
        G --> H{وضعیت پرداخت};
        H -- ناموفق --> F;
        H -- موفق --> I(ثبت موقت رزرو در سیستم);
    end
    subgraph سیستم و مدیر
        I --> J(سیستم: ثبت رزرو موقت);
        J --> K(مدیر: دریافت اعلان رزرو جدید);
        K --> L(مدیر: تخصیص نیروی خدماتی);
        L --> M(سیستم: به‌روزرسانی وضعیت رزرو به 'تأیید شده');
        M --> N(ارسال اعلان به مشتری و نیروی خدماتی);
    end
    N --> O(مشتری: مشاهده وضعیت 'تأیید شده');
    N --> P(نیروی خدماتی: مشاهده رزرو جدید);
    O --> Q(پایان فرآیند رزرو اولیه);
